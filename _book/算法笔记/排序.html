2-1：选择排序

public class Selection {
    public static void sort(int[] a) {
        for (int i = 0; i < a.length; i++) {
            int min = i; // 记录最小值的下标
            for (int j = i + 1; j < a.length; j++) {
                if (a[j] < a[min] ) {
                    min = j;
                }
            }
            int temp = a[min];
            a[min] = a[i];
            a[i] = temp;
        }
    }
}

class Solution {
    public void sortColors(int[] nums) {
        int l = -1 ,i = 0 ,h = nums.length;
        while (i < h) {
            if (nums[i] == 0) {
                swap(nums, ++l , i++);
            }else if (nums[i] == 2) {
                swap(nums, i, --h );
            }else{
                i++;
            }
        }
    }
    public void swap(int[] nums, int i , int j) {
        int tmp = nums[i];
        nums[i] = nums[j];
        nums[j] = tmp;
    }
}
